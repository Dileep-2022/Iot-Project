#include <Servo.h> //includes the servo library
#include <Wire.h> 
#include <LiquidCrystal_I2C.h>
LiquidCrystal_I2C lcd(0x27, 16, 2);
Servo myservo;
#define ir_enter 2
#define ir_back  4
#define ir_car1 5
#define ir_car2 6
#define ir_car3 7
#define ir_car4 8
bool initialDisplay = true;

int S1=0, S2=0, S3=0, S4=0;
int flag1=0, flag2=0; 
int slot = 4;  
void setup(){
	Serial.begin(9600);
	pinMode(ir_car1, INPUT);
	pinMode(ir_car2, INPUT);
	pinMode(ir_car3, INPUT);
	pinMode(ir_car4, INPUT);
	pinMode(ir_enter, INPUT);
	pinMode(ir_back, INPUT);

	myservo.attach(3);
	myservo.write(80);
	lcd.init(); // Initialize the LCD
	lcd.backlight(); // Turn on backlight
	lcd.setCursor(0, 0);
	lcd.print("  Have Slot: ");
	lcd.print(slot);
	lcd.setCursor(1, 1);
	lcd.print("Smart  Parking");

	delay(3000);
	lcd.clear();
  
	Read_Sensor();
}
void loop(){
	Read_Sensor();

	if(s1=s2=s3=s4==1){
	flag1=1;
	}

if (initialDisplay) {
    lcd.setCursor(0, 0);
    lcd.print("  Have Slot: ");
    lcd.print(slot);
    lcd.print("    ");
    initialDisplay = false;
    lcd.clear();
  }


lcd.setCursor (0,0);
if(S1==1){lcd.print("S1:Fill ");}
     else{lcd.print("S1:Empty");}
lcd.setCursor (8,0);
if(S2==1){lcd.print("S2:Fill ");}
     else{lcd.print("S2:Empty");}
lcd.setCursor (0,1);
if(S3==1){lcd.print("S3:Fill ");}
     else{lcd.print("S3:Empty");}
lcd.setCursor (8,1);
if(S4==1){lcd.print("S4:Fill ");}
     else{lcd.print("S4:Empty");}

if(digitalRead (ir_enter) == 1 && flag1==0){
myservo.write(180);
}
else {
lcd.clear();
  lcd.setCursor (0,0);
  lcd.print(" Sorry Parking ");  
  lcd.setCursor (0,1);
  lcd.print(" Full ");
  delay(1500);
lcd.clear();
}
if(digitalRead (ir_back) == 1 && flag2==0){
myservo.write(180);
}
}

void Read_Sensor(){
S1=0, S2=0, S3=0, S4=0;
if(digitalRead(ir_car1) == 0){S1=1;}
if(digitalRead(ir_car2) == 0){S2=1;}
if(digitalRead(ir_car3) == 0){S3=1;}
if(digitalRead(ir_car4) == 0){S4=1;}
  
}
